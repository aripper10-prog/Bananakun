// ...（録音部分の抜粋）... async function sendToServerStreaming(blob) { msg.innerText = "ストリーミング変換中..."; // チャンクに分けて送信するか、すてる君のライブラリがあるならそれを流用 const response = await fetch('/api/transform-stream', { method: 'POST', headers: { 'Content-Type': 'audio/webm' }, body: blob // 直接Blobを投げ、サーバー側でストリームとして受ける }); const reader = response.body.getReader(); const decoder = new TextDecoder(); bananaScript = ""; while (true) { const { done, value } = await reader.read(); if (done) break; const chunk = decoder.decode(value, { stream: true }); // ストリーミングで返ってくるバナナ語を逐次表示 bananaScript += chunk; msg.innerText = "生成中: " + bananaScript; } playBtn.classList.add('ready-to-play'); }
